\chapter{Methods}
\label{ch:methods}

\begin{equation}
	\mu = \frac{BR^3}{2}
\end{equation}

\begin{equation}
	\dot{E} = -\frac{2\mu^2\omega^4 \sin^2 \!\chi}{3c^3}
\end{equation}

\begin{equation}
	\dot{E} = -\frac{\mu^2\omega^4 \left(1 + \sin^2 \!\chi \right)}{c^3}
\end{equation}

\begin{equation}
	\tau \equiv -\frac{E}{\dot{E}}
\end{equation}

\begin{equation}
	I = \frac{2 M R^2}{5}
\end{equation}

\begin{equation}
	E = \frac{I\omega^2}{2}
\end{equation}

\begin{equation}
	\dot{E} = I \omega\dot{\omega} = K \omega^4
\end{equation}

\begin{equation}
	I \dot{\omega} = K \omega^3
\end{equation}

\begin{equation}
	I \omega^{-3} d\omega = K dt
\end{equation}

\begin{equation}
	\omega^{-2} = \omega_0^{-2} - \frac{2Kt}{I} = \omega_0^{-2} \left( 1 + \frac{t}{\tau} \right)
\end{equation}

\begin{equation}
	\tau = - \frac{I}{2K\omega^2}
\end{equation}

\begin{equation}
	\omega = \frac{\omega_0}{\sqrt{1 + \frac{t}{\tau}}}
\end{equation}
